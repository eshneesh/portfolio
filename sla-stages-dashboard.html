Расчеты DAX

Меры для SLA-модели в Power BI

Ниже — шаблон мер. Названия таблиц и колонок можно заменить под вашу модель. Для SLA в рабочих часах лучше хранить рассчитанные business-minutes в факте этапов.

Ожидаемая модель: `Tickets` — заявка; `TicketStages` — история этапов; `Stages` — справочник этапов; `Calendar` — календарь; `SLA Rules` — нормативы по приоритету, типу и этапу.
Ключевые поля: `TicketID`, `CreatedAt`, `ClosedAt`, `FirstResponseAt`, `DueAt`, `CurrentStage`, `Priority`, `Owner`, `Status`, `StageStartedAt`, `StageEndedAt`, `StageSlaMinutes`, `BusinessMinutes`.
Базовые меры заявок и SLA 
Total Tickets =
DISTINCTCOUNT ( Tickets[TicketID] )

Open Tickets =
CALCULATE (
    [Total Tickets],
    ISBLANK ( Tickets[ClosedAt] )
)

Closed Tickets =
CALCULATE (
    [Total Tickets],
    NOT ISBLANK ( Tickets[ClosedAt] )
)

Tickets In SLA =
CALCULATE (
    [Closed Tickets],
    Tickets[ClosedAt] <= Tickets[DueAt]
)

Tickets Breached =
CALCULATE (
    [Closed Tickets],
    Tickets[ClosedAt] > Tickets[DueAt]
)

Open Breached =
CALCULATE (
    [Open Tickets],
    NOW () > Tickets[DueAt]
)

SLA % =
DIVIDE ( [Tickets In SLA], [Closed Tickets] )

Breach % =
DIVIDE ( [Tickets Breached], [Closed Tickets] )

Open Backlog % =
DIVIDE ( [Open Tickets], [Total Tickets] )
Риск, остаток SLA и просрочка 
Remaining SLA Minutes =
SUMX (
    FILTER ( Tickets, ISBLANK ( Tickets[ClosedAt] ) ),
    DATEDIFF ( NOW (), Tickets[DueAt], MINUTE )
)

Avg Remaining SLA Minutes =
AVERAGEX (
    FILTER ( Tickets, ISBLANK ( Tickets[ClosedAt] ) ),
    DATEDIFF ( NOW (), Tickets[DueAt], MINUTE )
)

Breach Minutes =
SUMX (
    Tickets,
    VAR EndTime = COALESCE ( Tickets[ClosedAt], NOW () )
    RETURN
        MAX ( 0, DATEDIFF ( Tickets[DueAt], EndTime, MINUTE ) )
)

Avg Breach Minutes =
AVERAGEX (
    Tickets,
    VAR EndTime = COALESCE ( Tickets[ClosedAt], NOW () )
    RETURN
        MAX ( 0, DATEDIFF ( Tickets[DueAt], EndTime, MINUTE ) )
)

At Risk Tickets =
VAR RiskWindowMinutes = 120
RETURN
CALCULATE (
    [Open Tickets],
    FILTER (
        Tickets,
        VAR MinutesLeft = DATEDIFF ( NOW (), Tickets[DueAt], MINUTE )
        RETURN MinutesLeft >= 0 && MinutesLeft <= RiskWindowMinutes
    )
)

Due Today =
CALCULATE (
    [Open Tickets],
    DATEVALUE ( Tickets[DueAt] ) = TODAY ()
)

No Owner Tickets =
CALCULATE (
    [Open Tickets],
    ISBLANK ( Tickets[Owner] )
)
Время реакции, решения и перцентили 
Avg First Response Minutes =
AVERAGEX (
    FILTER ( Tickets, NOT ISBLANK ( Tickets[FirstResponseAt] ) ),
    DATEDIFF ( Tickets[CreatedAt], Tickets[FirstResponseAt], MINUTE )
)

First Response SLA % =
DIVIDE (
    CALCULATE (
        [Total Tickets],
        Tickets[FirstResponseAt] <= Tickets[FirstResponseDueAt]
    ),
    CALCULATE (
        [Total Tickets],
        NOT ISBLANK ( Tickets[FirstResponseAt] )
    )
)

Avg Resolution Hours =
AVERAGEX (
    FILTER ( Tickets, NOT ISBLANK ( Tickets[ClosedAt] ) ),
    DIVIDE ( DATEDIFF ( Tickets[CreatedAt], Tickets[ClosedAt], MINUTE ), 60 )
)

Median Resolution Hours =
MEDIANX (
    FILTER ( Tickets, NOT ISBLANK ( Tickets[ClosedAt] ) ),
    DIVIDE ( DATEDIFF ( Tickets[CreatedAt], Tickets[ClosedAt], MINUTE ), 60 )
)

P90 Resolution Hours =
PERCENTILEX.INC (
    FILTER ( Tickets, NOT ISBLANK ( Tickets[ClosedAt] ) ),
    DIVIDE ( DATEDIFF ( Tickets[CreatedAt], Tickets[ClosedAt], MINUTE ), 60 ),
    0.9
)

P95 Resolution Hours =
PERCENTILEX.INC (
    FILTER ( Tickets, NOT ISBLANK ( Tickets[ClosedAt] ) ),
    DIVIDE ( DATEDIFF ( Tickets[CreatedAt], Tickets[ClosedAt], MINUTE ), 60 ),
    0.95
)

Avg Open Ticket Age Hours =
AVERAGEX (
    FILTER ( Tickets, ISBLANK ( Tickets[ClosedAt] ) ),
    DIVIDE ( DATEDIFF ( Tickets[CreatedAt], NOW (), MINUTE ), 60 )
)
Метрики этапов, WIP и throughput 
Stage Tickets =
DISTINCTCOUNT ( TicketStages[TicketID] )

Current Stage Tickets =
CALCULATE (
    DISTINCTCOUNT ( Tickets[TicketID] ),
    NOT ISBLANK ( Tickets[CurrentStage] ),
    ISBLANK ( Tickets[ClosedAt] )
)

Avg Stage Duration Minutes =
AVERAGEX (
    TicketStages,
    VAR EndTime = COALESCE ( TicketStages[StageEndedAt], NOW () )
    RETURN DATEDIFF ( TicketStages[StageStartedAt], EndTime, MINUTE )
)

Stage SLA % =
DIVIDE (
    CALCULATE (
        DISTINCTCOUNT ( TicketStages[TicketID] ),
        TicketStages[BusinessMinutes] <= TicketStages[StageSlaMinutes]
    ),
    DISTINCTCOUNT ( TicketStages[TicketID] )
)

Stage Breached Tickets =
CALCULATE (
    DISTINCTCOUNT ( TicketStages[TicketID] ),
    TicketStages[BusinessMinutes] > TicketStages[StageSlaMinutes]
)

WIP Tickets =
CALCULATE (
    [Open Tickets],
    Tickets[CurrentStage] IN { "В работе", "Согласование", "Ожидание клиента" }
)

New Tickets =
CALCULATE (
    [Total Tickets],
    USERELATIONSHIP ( Tickets[CreatedDate], Calendar[Date] )
)

Closed Tickets by Close Date =
CALCULATE (
    [Closed Tickets],
    USERELATIONSHIP ( Tickets[ClosedDate], Calendar[Date] )
)

Throughput per Day =
DIVIDE (
    [Closed Tickets by Close Date],
    DISTINCTCOUNT ( Calendar[Date] )
)
Качество, эскалации, переоткрытия и aging buckets 
Reopened Tickets =
CALCULATE (
    [Total Tickets],
    Tickets[ReopenCount] > 0
)

Reopen Rate =
DIVIDE ( [Reopened Tickets], [Closed Tickets] )

Escalated Tickets =
CALCULATE (
    [Total Tickets],
    Tickets[EscalationCount] > 0
)

Escalation Rate =
DIVIDE ( [Escalated Tickets], [Total Tickets] )

Avg Touch Count =
AVERAGE ( Tickets[TouchCount] )

First Contact Resolution % =
DIVIDE (
    CALCULATE (
        [Closed Tickets],
        Tickets[ReopenCount] = 0,
        Tickets[EscalationCount] = 0
    ),
    [Closed Tickets]
)

No Update Over 4h =
CALCULATE (
    [Open Tickets],
    FILTER (
        Tickets,
        DATEDIFF ( Tickets[LastUpdatedAt], NOW (), HOUR ) >= 4
    )
)

Open 0-2h =
CALCULATE (
    [Open Tickets],
    FILTER ( Tickets, DATEDIFF ( Tickets[CreatedAt], NOW (), HOUR ) < 2 )
)

Open 2-8h =
CALCULATE (
    [Open Tickets],
    FILTER (
        Tickets,
        DATEDIFF ( Tickets[CreatedAt], NOW (), HOUR ) >= 2
            && DATEDIFF ( Tickets[CreatedAt], NOW (), HOUR ) < 8
    )
)

Open 8-24h =
CALCULATE (
    [Open Tickets],
    FILTER (
        Tickets,
        DATEDIFF ( Tickets[CreatedAt], NOW (), HOUR ) >= 8
            && DATEDIFF ( Tickets[CreatedAt], NOW (), HOUR ) < 24
    )
)

Open 24h+ =
CALCULATE (
    [Open Tickets],
    FILTER ( Tickets, DATEDIFF ( Tickets[CreatedAt], NOW (), HOUR ) >= 24 )
)

HTML, обернутый в DAX

Мини-визуалы для HTML Content / карточек

Эти меры возвращают HTML-строку. Их удобно использовать в Power BI через HTML Content visual или похожий визуал, где разрешен HTML/CSS.

Важно про безопасность и поддержку HTML-визуалов

HTML-визуалы зависят от конкретного кастомного визуала Power BI и настроек безопасности. Не вставляй в DAX пользовательский ввод без очистки. Для управляемых отчетов лучше ограничиваться простыми тегами: div, span, inline-style, без скриптов.

HTML-бейдж SLA 
SLA Badge HTML =
VAR Pct = [SLA %]
VAR Label =
    SWITCH (
        TRUE (),
        Pct >= 0.95, "В норме",
        Pct >= 0.90, "Наблюдать",
        "Риск"
    )
VAR Color =
    SWITCH (
        TRUE (),
        Pct >= 0.95, "#65ffc0",
        Pct >= 0.90, "#ffd166",
        "#ff6b8a"
    )
RETURN
    "<span style='display:inline-flex;align-items:center;padding:6px 10px;border-radius:8px;"
        & "border:1px solid " & Color & ";color:" & Color & ";"
        & "background:rgba(255,255,255,.04);font-weight:800'>"
        & Label & " · " & FORMAT ( Pct, "0.0%" )
        & "</span>"

Ticket Risk Row HTML =
VAR TicketId = SELECTEDVALUE ( Tickets[TicketID], "REQ-00000" )
VAR StageName = SELECTEDVALUE ( Tickets[CurrentStage], "Без этапа" )
VAR PriorityName = SELECTEDVALUE ( Tickets[Priority], "P3" )
VAR MinutesLeft =
    DATEDIFF ( NOW (), SELECTEDVALUE ( Tickets[DueAt] ), MINUTE )
VAR StatusText =
    SWITCH (
        TRUE (),
        MinutesLeft < 0, "Просрочено",
        MinutesLeft <= 120, "Риск",
        "В SLA"
    )

    SWITCH (
        TRUE (),

        "#65ffc0"
    )
RETURN
    "<div style='display:grid;grid-template-columns:90px 52px 1fr 100px;gap:10px;"
        & "align-items:center;padding:10px 12px;border-bottom:1px solid rgba(160,181,255,.14);"
        & "font-family:Segoe UI;color:#f7f9ff;background:#080b18'>"
        & "<strong>" & TicketId & "</strong>"
        & "<span style='color:#42e8ff;font-weight:800'>" & PriorityName & "</span>"
        & "<span style='color:#b9c4e8'>" & StageName & "</span>"
        & "<span style='justify-self:end;color:" & StatusColor & ";font-weight:800'>" & StatusText & "</span>"
        & "</div>"
HTML-панель топ-этапов по риску 
Top Risk Stages HTML =
VAR TopStages =
    TOPN (
        5,
        ADDCOLUMNS (
            VALUES ( Stages[StageName] ),
            "RiskTickets", [At Risk Tickets] + [Open Breached]
        ),
        [RiskTickets], DESC
    )
RETURN
    "<div style='font-family:Segoe UI;color:#f7f9ff;width:100%'>"
        & CONCATENATEX (
            TopStages,
            VAR StageName = Stages[StageName]
            VAR RiskTickets = [RiskTickets]
            VAR WidthPct = FORMAT ( DIVIDE ( RiskTickets, MAXX ( TopStages, [RiskTickets] ) ), "0%" )
            RETURN
                "<div style='margin-bottom:10px'>"
                    & "<div style='display:flex;justify-content:space-between;margin-bottom:5px;font-size:12px;color:#b9c4e8'>"
                    & "<span>" & StageName & "</span><strong style='color:#f7f9ff'>" & FORMAT ( RiskTickets, "#,0" ) & "</strong></div>"
                    & "<div style='height:9px;background:#14182d;border-radius:999px;overflow:hidden'>"
                    & "<div style='height:100%;width:" & WidthPct & ";background:linear-gradient(90deg,#42e8ff,#9c64ff,#ff58ca)'></div>"
                    & "</div></div>",
            ""
        )
        & "</div>"
