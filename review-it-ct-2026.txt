        const shouldFormatValues = options.formatValues !== false;
        const deptValues = new Map();
        $$("#departmentKpi tbody tr").forEach((row) => {
          const inverse = isInverseRow(row);
          const since2026 = row.dataset.since === "2026";
          const get = (className) => parseNum($(`.${className}`, row)?.textContent);

          const q1 = completion(get("q1-plan"), get("q1-fact"), inverse);
          const q2 = completion(get("q2-plan"), get("q2-fact"), inverse);
          const h26 = completion(get("h26-plan"), get("h26-fact"), inverse);
          const h25 = since2026 ? NaN : completion(get("h25-plan"), get("h25-fact"), inverse);
          const delta = Number.isFinite(h26) && Number.isFinite(h25) ? h26 - h25 : NaN;

          [
            ["q1-pct", q1],
            ["q2-pct", q2],
            ["h26-pct", h26],
            ["h25-pct", h25],
            ["delta", delta]
          ].forEach(([className, value]) => {
            const cell = $(`.${className}`, row);
            if (!cell) return;
            cell.textContent = className === "delta" && Number.isFinite(value)
              ? `${value > 0 ? "+" : ""}${Math.round(value)} п.п.`
              : fmtPct(value);
            setPctClass(cell, className === "delta" ? 100 + value : value);
          });

          const dept = row.dataset.dept || row.cells[0]?.textContent.trim();
          if (dept && Number.isFinite(h26)) {
            if (!deptValues.has(dept)) deptValues.set(dept, []);
            deptValues.get(dept).push(h26);
          }

          if (shouldFormatValues) formatMoneyCells(row);
        });

        const averages = {};
        deptValues.forEach((values, dept) => {
          averages[dept] = values.reduce((sum, value) => sum + value, 0) / values.length;
        });

        updateBars(averages);
        updateSpark(averages);
        const totalValues = Object.values(averages);
        const total = totalValues.length ? totalValues.reduce((sum, value) => sum + value, 0) / totalValues.length : 0;
        $$("[data-summary-kpi], [data-summary-kpi-ring]").forEach((node) => {
          node.textContent = fmtPct(total);
        });
      }

      function updateBars(averages) {
        $$("#deptBars .bar-row").forEach((row) => {
          const dept = row.dataset.bar;
          const value = averages[dept] || 0;
          $(".bar-fill", row).style.setProperty("--w", `${Math.min(value, 125)}%`);
          $(".bar-value", row).textContent = fmtPct(value);
        });
      }

      function updateSpark(averages) {
        const depts = ["Продажи", "Техпресейл", "Поддержка"];
        const points = depts.map((dept, index) => {
          const x = [52, 177, 302][index];
          const value = averages[dept] || 0;
          const y = 150 - Math.max(0, Math.min(125, value)) * 0.88;
          return { x, y, value };
        });
        $("#sparkLine").setAttribute("points", points.map((p) => `${p.x},${p.y}`).join(" "));
        $("#sparkPoints").innerHTML = points.map((p) => `
          <g>
            <circle cx="${p.x}" cy="${p.y}" r="7" stroke-width="4"></circle>
            <text x="${p.x}" y="${p.y - 14}" text-anchor="middle" font-size="12" font-weight="900">${fmtPct(p.value)}</text>
          </g>
        `).join("");
      }

      function loadEmbeddedFile(file) {
        if (!file) return;
        const isPdf = file.type === "application/pdf" || /\.pdf$/i.test(file.name);
        const isImage = file.type.startsWith("image/");
        if (!isPdf && !isImage) {
          showToast("Поддерживаются PDF и изображения");
          return;

          kind: isPdf ? "pdf" : "image",

        preview.replaceChildren(createFileViewerElement(false));
        $("[data-file-name]").textContent = embeddedFile.name;
        $("[data-file-type]").textContent = embeddedFile.type;

      async function assetExists(url) {
        if (!url || url.startsWith("blob:")) return Boolean(url);
        try {
          const response = await fetch(url, { method: "HEAD", cache: "no-store" });
          return response.ok;
        } catch {
          return false;
        }
      }

      async function openExcelShot(button) {
        const key = button.dataset.shotKey;
        const saved = key ? excelShots.get(key) : null;
        if (saved?.url) {
          openExcelShotPreview(saved.title, saved.url);

        showToast("Выберите PNG/JPG скриншот отчета");
        selectExcelShot(button);
      }

      function loadExcelShot(input) {
        const file = input.files?.[0];
        if (!file) return;
        if (!file.type.startsWith("image/")) {
          showToast("Для скриншота нужна картинка");
          input.value = "";
          return;
        }

        const key = input.dataset.excelShotInput;
        const previous = excelShots.get(key);
        if (previous?.url?.startsWith("blob:")) URL.revokeObjectURL(previous.url);
        const shot = {
          title: input.dataset.shotTitle || file.name,
          url: URL.createObjectURL(file)
        };
        excelShots.set(key, shot);
        openExcelShotPreview(shot.title, shot.url);
        showToast("Скриншот добавлен");
      }

      function filterAnalytics() {
        const analytics = $("#analytics");
        if (!analytics) return;

        const search = analytics.querySelector("[data-analytics-search]");
        const query = (search?.value || "").trim().toLowerCase();
        const activeFilter = analytics.querySelector("[data-analytics-filter].active")?.dataset.analyticsFilter || "all";
        let visibleCards = 0;

        analytics.querySelectorAll("[data-analytics-card]").forEach((card) => {
          const module = card.closest("[data-analytics-module]");
          const moduleName = module?.dataset.analyticsModule || "";
          const haystack = `${card.dataset.search || ""} ${card.textContent}`.toLowerCase();
          const matchesModule = activeFilter === "all" || moduleName === activeFilter;
          const matchesQuery = !query || haystack.includes(query);
          const visible = matchesModule && matchesQuery;
          card.classList.toggle("is-hidden", !visible);
          if (visible) visibleCards += 1;
        });

        analytics.querySelectorAll("[data-analytics-module]").forEach((module) => {
          const hasVisibleCard = module.querySelector("[data-analytics-card]:not(.is-hidden)");
          module.classList.toggle("is-hidden", !hasVisibleCard);
        });

        $("#analyticsNoResults")?.classList.toggle("is-visible", visibleCards === 0);
      }

      function activateSection(id, options = {}) {
        const section = document.getElementById(id);
        if (!section || !section.matches("[data-section]")) return;

        $$("[data-section]").forEach((item) => {
          const active = item === section;
          item.classList.toggle("active-section", active);
          item.setAttribute("aria-hidden", active ? "false" : "true");
        });

        $$(".nav-link").forEach((link) => {
          link.classList.toggle("active", link.getAttribute("href") === `#${id}`);
        });

        if (!options.skipHash) {
          history.replaceState(null, "", `#${id}`);
        }

        if (!options.keepScroll) {
          window.scrollTo({ top: 0, behavior: "smooth" });
        }
      }

      document.addEventListener("input", (event) => {
        // Добавлено: пересчитываем KPI на лету, но не превращаем введенное число в "млн руб." прямо во время набора.
        if (event.target.closest("#departmentKpi")) recalcKPIs({ formatValues: false });
        if (event.target.matches("[data-analytics-search]")) filterAnalytics();
      });

      document.addEventListener("focusout", (event) => {
        // Добавлено: после выхода из KPI-ячейки применяем красивый формат денежных значений.
        if (event.target.closest("#departmentKpi")) recalcKPIs();
      });

      document.addEventListener("change", (event) => {
        const target = event.target;
        if (target.matches("[data-file-input]")) {
          loadEmbeddedFile(target.files?.[0]);
        }
        if (target.matches("[data-excel-shot-input]")) {
          loadExcelShot(target);
        }
      });

      document.addEventListener("click", async (event) => {
        const target = event.target.closest("button, a");
        if (!target) return;

        if (target.matches("[data-toggle-edit]")) {
          body.classList.toggle("editing");
          target.textContent = body.classList.contains("editing") ? "Редактирование включено" : "Режим редактирования";
          showToast(body.classList.contains("editing") ? "Поля подсвечены для правок" : "Редактирование выключено");
        }

        if (target.matches("[data-toggle-theme]")) {
          setTheme(body.classList.contains("theme-dark") ? "light" : "dark");
          showToast(body.classList.contains("theme-dark") ? "Включена темная тема" : "Включена светлая тема");
        }

        if (target.matches(".nav-link")) {
          event.preventDefault();
          const id = target.getAttribute("href")?.replace("#", "");
          if (id) activateSection(id);
        }

        if (target.matches("[data-recalc]")) {
          recalcKPIs();
          showToast("KPI и графики пересчитаны");
        }

        if (target.matches("[data-print]")) {
          window.print();
        }

        if (target.matches("[data-copy-summary]")) {
          const text = $("#executiveSummary")?.textContent.trim() || "";
          try {
            await navigator.clipboard.writeText(text);
            showToast("Краткий вывод скопирован");
          } catch {
            showToast("Краткий вывод готов к копированию");
          }
        }

        if (target.matches("[data-copy-path]")) {
          try {
            await navigator.clipboard.writeText(target.dataset.copyPath || "");
            showToast("Путь скопирован");
          } catch {
            showToast("Не удалось скопировать путь");
          }
        }

        if (target.matches("[data-kpi-filter]")) {
          filterKpi(target.dataset.kpiFilter);
        }

        if (target.matches("[data-analytics-filter]")) {
          $$("#analytics [data-analytics-filter]").forEach((button) => {
            button.classList.toggle("active", button === target);
          });
          filterAnalytics();
          $("#analytics .analytics-toolbar")?.scrollIntoView({ block: "start", behavior: "smooth" });
        }

        if (target.matches("[data-add-kpi]")) {
          addKpiRow();
        }

        if (target.matches("[data-toggle-employees]")) {
          body.classList.toggle("compact-employees");
          target.textContent = body.classList.contains("compact-employees") ? "План/факт" : "Только проценты";
        }

        if (target.matches("[data-open-report]")) {
          openModal(target.dataset.openReport);
        }

        if (target.matches("[data-excel-download]")) {
          if (location.protocol !== "file:") {
            event.preventDefault();
            await downloadExcelAsset(target);
          }
        }

        if (target.matches("[data-open-excel-shot]")) {
          await openExcelShot(target);
        }

        if (target.matches("[data-optics-view]")) {
          switchOpticsView(target.dataset.opticsView);
        }

        if (target.matches("[data-open-file-full]")) {
          openEmbeddedFileFull();
        }

        if (target.matches("[data-clear-file]")) {
          resetEmbeddedFile();
          showToast("Файл очищен");
        }

        if (target.matches("[data-close-modal]")) {
          closeModal();
        }
      });

      $("#reportModal").addEventListener("click", (event) => {
        if (event.target.id === "reportModal") closeModal();
      });

      document.addEventListener("keydown", (event) => {
        if (event.key === "Escape") closeModal();
      });

      window.addEventListener("hashchange", () => {
        const id = location.hash.replace("#", "") || "overview";
        activateSection(id, { skipHash: true });
      });

      initTheme();
      recalcKPIs();
      filterAnalytics();
      activateSection(location.hash.replace("#", "") || "overview", { skipHash: true, keepScroll: true });
    })();
  </script>
</body>
</html>
